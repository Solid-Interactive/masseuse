* added a riveted view
    * to use (also see [tests/plugins/rivets/viewTest.js](../tests/plugins/rivets/viewTest.js)

    ```html
    <div id="blah">
        <div>{{data.title}}</div>
    </div>
    ```

    ```javascript
    RivetView = rivetPlugins.view;
    new RivetView({
          el : '#blah',
          templateHtml : template,
          modelData : {
              title : 'There it is.'
          }
      }).start();
    ```

    ```html
    <div id="blah">
        <div>There it is.</div>
    </div>
    ```
